function pow( x : real, n : int ) : real
{
    let y : real = 4.0;
    if (n>0){
        while (n>0){
            set y <- y * x;
            set n <- n - 1;
        }
    }else{
            while (n<0){
                set y <- y / x;
                set n <- n + 1;
            }
    }
    y;
}
